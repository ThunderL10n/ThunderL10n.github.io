Enkel C++ kalkulator

#include <iostream>
using namespace std;
int main()
{
int valg;
float tall1;
float tall2;
int heltall1;
int heltall2;
while (valg != 7) {
cout <<"velg regne oprasjon: \n1 = +\n2 = -\n3 = *\n4 = /\n5 = heltallsdivisjon\n6 = %\n7 = Exit\n";
cin >> valg;

if( valg > 0 && valg != 6 && valg != 5 && valg != 7 && valg < 7){
cout << "skriv inn to tall: \n";
cin >> tall1 >> tall2;
}
else if (valg > 0 && valg == 6 || valg == 5 && valg < 7){
cout << "skriv inn to heltall: \n";
cin >> heltall1 >> heltall2;
}
switch (valg) {
case 1:
cout << tall1 << " + " << tall2 << " = " <<"\033[21m"<< tall1 + tall2 <<"\033[0m"<< endl;
break;
case 2:
cout << tall1 << " - " << tall2 << " = " <<"\033[21m"<< tall1 - tall2<<"\033[0m" << endl;
break;
case 3:
cout << tall1 << " * " << tall2 << " = " <<"\033[21m"<< tall1 * tall2 <<"\033[0m"<< endl;
break;
case 4:
if (tall2 == 0){
cout << "Kan ikke dele på 0 \n";
}
else{
cout << tall1 << " / " << tall2 << " = " <<"\033[21m"<< tall1 / tall2 <<"\033[0m"<< endl;
}
break;
case 5:
if (heltall2 == 0){
cout << "Kan ikke dele på 0 \n";
}
else{
cout << heltall1 << " / " << heltall2 << " = " <<"\033[21m"<< heltall1 / heltall2 <<"\033[0m"<< endl;
}
break;
case 6:
if (heltall2 == 0){
cout << "Kan ikke gjøre modulus med 0 \n";
}
else{
cout << heltall1 << " % " << heltall2 << " = " <<"\033[21m"<< heltall1 % heltall2 <<"\033[0m"<< endl;
}
break;
case 7:
cout << "Avslutter kalkulator \n";
break;
default:
cout << "ugyldig valg, prøv igjen \n";
break;

}
}

return 0;
system("pause>0");
}
